make_lowercase <- function(df){
  for (i in 1:length(names(df))){
    names(df)[i] <- tolower(names(df)[i])
  }
  return(df)
}

get_spec <- function(spec_code, orig_df){
  dat <- orig_df %>%
    filter(species_code==spec_code)
  return(dat)
}
